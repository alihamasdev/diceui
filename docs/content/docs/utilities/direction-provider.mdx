## Usage

```tsx
import { DirectionProvider } from "@diceui/direction-provider"

export default function App() {
  return (
    <DirectionProvider defaultDirection="ltr">
      <YourApp />
    </DirectionProvider>
  )
}
```

## API Reference

### DirectionProvider

The main provider component that manages direction context.

#### Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `defaultDirection` | `"ltr" \| "rtl"` | `"ltr"` | The default text direction |
| `children` | `ReactNode` | - | The child components |

### useDirection

A hook to access the current direction.

```tsx
import { useDirection } from "@diceui/direction-provider"


function Component() {
  const { direction} = useDirection()
  
  return (
    <button dir={direction}>
      Text direction
    </button>
  )
}
```
